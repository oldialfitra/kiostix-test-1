CREATE TABLE IF NOT EXISTS Penulis (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        name VARCHAR(255)
    );

CREATE TABLE IF NOT EXISTS BUKU(
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name VARCHAR(255),
    PenulisId INTEGER,
    FOREIGN KEY (PenulisId) REFERENCES Penulis(id)
            ON DELETE SET NULL
            ON UPDATE CASCADE);

CREATE TABLE IF NOT EXISTS BUKUKATEGORI(
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    BukuId INTEGER,
    KategoriId INTEGER,
    FOREIGN KEY (BukuId) REFERENCES Buku(id)
            ON DELETE SET NULL
            ON UPDATE CASCADE
    FOREIGN KEY (KategoriId) REFERENCES Kategori(id)
            ON DELETE SET NULL
            ON UPDATE CASCADE););

CREATE TABLE IF NOT EXISTS Kategori(
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name VARCHAR(255));
